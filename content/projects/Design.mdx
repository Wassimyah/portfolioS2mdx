// Variabele voor tellen
let count = 0;

// Functie voor het genereren van kleuren
function generateComplementaryColor() {
    // Willekeurige kleur maken
    const randomColor = '#' + Math.floor(Math.random()*16777215).toString(16);

    // Complementaire kleur berekenen
    const hex = randomColor.substring(1);
    const r = parseInt(hex.substring(0, 2), 16);
    const g = parseInt(hex.substring(2, 4), 16);
    const b = parseInt(hex.substring(4, 6), 16);
    const compR = 255 - r;
    const compG = 255 - g;
    const compB = 255 - b;
    const complementaryColor = '#' + (compR * 65536 + compG * 256 + compB).toString(16).padStart(6, '0');

    // Teruggeven van kleuren
    return { randomColor, complementaryColor };
}

// Functie voor het wijzigen van kleuren en tellen
function changeColorsAndCount() {
    // Elementen ophalen
    const countingText = document.getElementById("countingText");
    const countingNumber = document.getElementById("countingNumber");

    // Kleuren wijzigen
    const { randomColor, complementaryColor } = generateComplementaryColor();
    countingText.style.color = complementaryColor;
    countingText.style.backgroundColor = 'transparent';
    countingNumber.style.color = randomColor;
    countingNumber.style.backgroundColor = complementaryColor;

    // Tellen
    count++;
    countingNumber.textContent = count;
}

// Periodiek kleuren wijzigen
setInterval(changeColorsAndCount, 1000);
